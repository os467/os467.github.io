BP神经网络

这是一个简单的神经网络

import numpy as np

class OurNeuralNetWork:
    '''
    神经网络：
    具有两个输入值
    一个隐藏层 和两个神经元
    一个输出层和一个神经元
    每个神经元都有相同的权重和偏置
    '''
    def __init__(self):
        '''
        初始化神经网络
        '''
        weights = np.array([0,1])
        bias = 0
        
        #初始化三个神经元节点
        self.h1 = Neuron(weights,bias)
        self.h2 = Neuron(weights,bias)
        self.o1 = Neuron(weights,bias)
        
    def feedforward(self,x):
        '''
        神经网络的前馈行为
        向h1和h2神经元输入数据
        '''
        out_h1 = self.h1.feedforward(x)
        print(out_h1)
        out_h2 = self.h2.feedforward(x)
        
        out_o1 = self.o1.feedforward(np.array([out_h1,out_h2]))       
        return out_o1
    
def sigmoid(x):
    return 1/(1+np.exp(-x))
    
class Neuron:
    def __init__(self,weights,bias):
        '''
        初始化，神经元的权重和偏置
        '''
        self.weights = weights
        self.bias = bias
        
    def feedforward(self,x):
        '''
        前馈函数
        x为输入
        '''
        #向量计算
        total = np.dot(self.weights,x) +self.bias
        #激活函数
        return sigmoid(total)

network = OurNeuralNetWork()

#所有输入节点
x=np.array([2,3])

print(network.feedforward(x))

损失差函数

用于计算 预判特征向量和真实特征向量间存在的误差估计
$$
MSE = {1\over{n}}\sum_{i=1}^n(y_{true}-y_{pred})^2
$$

这个神经网络不够好，还要不断优化，尽量减少损失。我们知道，改变网络的权重和偏置可以影响预测值，但我们应该怎么做呢？

预测值是由一系列网络权重和偏置计算出来的：

所以损失函数实际上是包含多个权重、偏置的多元函数
$$
L(w_1,w_2,w_3,w_4,w_5,w_6,b_1,b_2,b_3)
$$
如果调整一下w1，损失函数是会变大还是变小？我们需要知道偏导数∂L/∂w1是正是负才能回答这个问题。

根据链式求导法则：
$$
{∂L\over∂w_1} = {∂L\over∂y_{pred}} * {∂y_{pred}\over∂w_1}
$$

而L=(1-ypred)^2，可以求得第一项偏导数：
$$
{∂L\over∂y_{pred}} = {(∂{1-y_{pred}})^2\over∂y_{pred}} = -2(1-y_{pred})
$$

而y_pred = o1

随机梯度下降

一开始的w和b是随机生成的

η是一个常数，称为学习率（learning rate），它决定了我们训练网络速率的快慢。将w1减去η·∂L/∂w1，就等到了新的权重w1。

当∂L/∂w1是正数时，w1会变小；当∂L/∂w1是负数 时，w1会变大。

如果我们用这种方法去逐步改变网络的权重w和偏置b，损失函数会缓慢地降低，从而改进我们的神经网络

反向传播
$$
w_{1_{new}} = w_{1} - η{∂L\over∂w_1}
$$

小案例

根据真实数据推测人的性别

具有两个输入参数 weight 和 height

import numpy as np

class OurNeuralNetWork:
    '''
    神经网络：
    具有两个输入值
    一个隐藏层 和两个神经元
    一个输出层和一个神经元
    每个神经元都有相同的权重和偏置
    '''
    def __init__(self):
        '''
        初始化神经网络
        '''
        
        #weights
        self.w1 = np.random.normal();
        self.w2 = np.random.normal();
        self.w3 = np.random.normal();
        self.w4 = np.random.normal();
        self.w5 = np.random.normal();
        self.w6 = np.random.normal();
        
        #biases
        self.b1 = np.random.normal();
        self.b2 = np.random.normal();
        self.b3 = np.random.normal();

        
    def feedforward(self,x):
        '''
        神经网络的前馈行为
        向h1和h2神经元输入数据
        x是一个具有两个元素的特征向量
        '''
        h1 = sigmoid(self.w1 * x[0] + self.w2 * x[1] + self.b1)
        h2 = sigmoid(self.w3 * x[0] + self.w4 * x[1] + self.b2)
        
        o1 = sigmoid(self.w5 * h1 + self.w6 * h2 + self.b3) 
        return o1
    
    def train(self,data,all_y_trues):
        #学习率
        learn_rate = 0.1
        #迭代次数
        epochs  = 3000
        
        #开始迭代
        for epoch in range(epochs):
            for x,y_true in zip(data,all_y_trues):
                #前馈
                sum_h1 = self.w1 * x[0] + self.w2 * x[1] + self.b1
                h1 = sigmoid(sum_h1)
                sum_h2 = self.w3 * x[0] + self.w4 * x[1] + self.b2
                h2 = sigmoid(sum_h2)
                
                sum_o1 = self.w5 * h1 + self.w6 * h2 + self.b3
                o1 = sigmoid(sum_o1) 
                y_pred = o1
                
                #梯度下降
                d_L_d_ypred = -2 * (y_true - y_pred)
                
                #Neuron o1
                d_ypred_d_w5 = h1 * deriv_sigmoid(sum_h1)
                d_ypred_d_w6 = h2 * deriv_sigmoid(sum_o1)
                d_ypred_d_b3 = deriv_sigmoid(sum_o1)
                
                d_ypred_d_h1 = self.w5 * deriv_sigmoid(sum_h1)
                d_ypred_d_h2 = self.w6 * deriv_sigmoid(sum_h2)
                
                #Neuron h1
                d_h1_d_w1 = x[0] * deriv_sigmoid(sum_h1)
                d_h1_d_w2 = x[1] * deriv_sigmoid(sum_h2)
                d_h1_d_b1 = deriv_sigmoid(sum_h2)
                
                #Neuron h2
                d_h2_d_w3 = x[0] * deriv_sigmoid(sum_h2)
                d_h2_d_w4 = x[1] * deriv_sigmoid(sum_h2)
                d_h2_d_b2 = deriv_sigmoid(sum_h2)
                
                #更新 权重 和 偏置
                # Neuron h1
                self.w1 -= learn_rate * d_L_d_ypred * d_ypred_d_h1 * d_h1_d_w1
                self.w2 -= learn_rate * d_L_d_ypred * d_ypred_d_h1 * d_h1_d_w2
                self.b1 -= learn_rate * d_L_d_ypred * d_ypred_d_h1 * d_h1_d_b1
        
                # Neuron h2
                self.w3 -= learn_rate * d_L_d_ypred * d_ypred_d_h2 * d_h2_d_w3
                self.w4 -= learn_rate * d_L_d_ypred * d_ypred_d_h2 * d_h2_d_w4
                self.b2 -= learn_rate * d_L_d_ypred * d_ypred_d_h2 * d_h2_d_b2
        
                # Neuron o1
                self.w5 -= learn_rate * d_L_d_ypred * d_ypred_d_w5
                self.w6 -= learn_rate * d_L_d_ypred * d_ypred_d_w6
                self.b3 -= learn_rate * d_L_d_ypred * d_ypred_d_b3
                
                if epoch % 10 == 0:
                    y_preds = np.apply_along_axis(self.feedforward, 1, data)
                    loss = mse_loss(all_y_trues, y_preds)
                    print("Epoch %d loss: %.3f" % (epoch, loss))
                
def deriv_sigmoid(x):
    '''
    对sigmoid函数求导
    '''
    fx = sigmoid(x)
    return fx * (1 - fx)    
    
def sigmoid(x):
    return 1/(1+np.exp(-x))
    

def mse_loss(y_true,y_pred):
    '''
    损失差函数
    '''
    return ((y_true - y_pred) ** 2).mean()



#正样本数据集  
data = np.array([
  [-2, -1],  # Alice
  [25, 6],   # Bob
  [17, 4],   # Charlie
  [-15, -6], # Diana
])

#1女,2男
all_y_trues = np.array([
  1, # Alice
  0, # Bob
  0, # Charlie
  1, # Diana
])

#训练神经网络
network = OurNeuralNetWork()
network.train(data, all_y_trues)

# Make some predictions
emily = np.array([-7, -3]) # 128 pounds, 63 inches
frank = np.array([20, 2])  # 155 pounds, 68 inches
print("Emily: %.3f" % network.feedforward(emily)) # 0.951 - F
print("Frank: %.3f" % network.feedforward(frank)) # 0.039 - M
